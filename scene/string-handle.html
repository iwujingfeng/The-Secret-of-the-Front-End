# 字符串处理

# 获取字符在字符串中的索引

const str = "hello world";
console.log("索引", str.indexOf("o")); // 4
console.log("索引", str.indexOf("o", 5)); // 7

案例解析

indexOf()方法用来获取一个字符在字符串中首次出现的位置的索引值。如果字符串中有重复字符，将只返回检索到的首次出现的索引 4。如果没有检索到该字符，返回索引-1。如果设置了第二个参数也就是开始检索的位置，检索到的首次出现的索引将是第二个 o，索引为 7。需要知道的是，数组和字符串一样，都有 indexOf()方法来获取一个元素在数组中的索引。

说明

数组和字符串一样，也有 indexOf()方法，用来获取一个元素在数组中的索引。

# 获取某个索引位置的字符

const str = "hello world";
console.log("charAt", str.charAt(4)); // 6

案例解析

charAt()方法用来获取指定索引位置的字符。

# 判断字符串中是否包含某个字符

const str = "hello world";
console.log("indexOf", str.indexOf("o") !== -1); // true
console.log("includes", str.includes("o")); // true
console.log("search", str.search("o") !== -1); // true
console.log("match", str.match(/o/));
// [ 'o', index: 4, input: 'hello world', groups: undefined ]
console.log("match", str.match(/o/)[0] === "o"); // true
console.log("test", /o/.test(str)); // true
console.log("exec", /o/.exec(str));
// [ 'o', index: 4, input: 'hello world', groups: undefined ]
console.log('exec', /o/.exec(str)[0] === 'o'

// 判断字符串开头或结尾是否包含某个字符
console.log("索引判断", str[0] === "h"); // true
console.log("索引判断", str.charAt(str.length - 1) === "d"); // true
console.log("startsWith", str.startsWith("h")); // true
console.log("endsWith", str.endsWith("d")); // true

案例解析

我们知道，indexOf()主要用来获取字符在字符串中首次出现的位置的索引值，但通过判断 !==-1，也可以用来判断该字符是否在字符串中存在。最为推荐的方法是 ES6 中增加的 includes()，从语义上，该方法就是判断字符是否包含在字符串中，返回值是布尔。还可以使用正则匹配来判断，使用字符串的 search()，如果匹配上，会返回匹配到的第一个字符的索引，如果没有匹配到，返回-1。match()方法会返回包含匹配到的字符的一个数组，可以通过判断该数组索引 0 位置的匹配字符是否和该字符相同来进行判断，如果没有匹配到，返回 null。正则对象的 test()如果能够匹配到该字符，返回 true，没有匹配到，返回 false。exec()方法的匹配结果和字符串方法 match()一样，所以依然可以通过判断该数组索引 0 位置的匹配字符是否和该字符相同来进行判断，如果没有匹配到，同样返回 null。

判断字符串开头和结尾是否包含某个字符，可以通过索引获取，索引 0 位置就是开头的字符，str.length - 1 是最后一个索引，通过 charAt()获取结尾的字符，判断是否和某个字符相同即可。最佳方法是使用 ES6 提供的 startsWith()方法判断字符串开头是否包含某个字符，endsWith()判断结尾是否包含某个字符，返回布尔值。

# 判断两个字符串是否严格相等

const str = "hello world";
const newStr = str;
console.log("Object.is", Object.is(str, newStr)); // true
console.log("===", str === newStr); // true
console.log("===", NaN === NaN); // false
console.log("===", +0 === -0); // true
console.log("Object.is", Object.is(NaN, NaN)); // true
console.log("Object.is", Object.is(+0, -0)); // false

案例解析

在判断值是否相等时，Object.is()是更加没有副作用的方法，和 === 一样，只有同类型数据才能比较，类型不同会被判断为 false，而==比较时会进行类型转换，关于类型转换，请参考核心技术-类型转换一章。Object.is()和===主要区别在于处理 NaN 是否等于 NaN，+0 是否等于-0 的情况。对于===，NaN 反而不等于 NaN，+0 反而等于-0，这是不严谨的，而 Object.is()会正确处理，NaN 等于 NaN，+0不等于-0。

# 字符串截取

const str = "hello world";
console.log("substring", str.substring(0, 5)); // hello
console.log("slice", str.slice(0, 5)); // hello
console.log("substr", str.substr(0, 5)); // hello
const strArr = str.split(" ");
console.log("split", strArr); // [ 'hello', 'world' ]
console.log("split", strArr[0]); // hello

// 去掉字符串的后两位，只截取前面的字符
const newStr = str.substring(0, str.length - 2);
console.log("字符截取", newStr);

案例解析

截取字符串中的字符，最常用的方法是 substring()、slice()和 substr()，前两个方法的两个参数是截取的开始索值和结尾索引值，但截取的结果不包括结尾索引位置的字符。substr()方法的两个参数是截取的开始索引值和截取的字符数。split()方法以一个分隔符（这里是空格" "）为边界，将字符串切成一个数组。通过索引来获取截取的字符。对于不知道要截取多少个字符，只想去掉后面两个或多个字符的情况，可以使用 substring()或 slice()方法，通过str.length - 2计算出去掉后面字符后要截取的结尾的索引，完成不定数字符的截取。

说明

数组和字符串一样，也有 slice()方法，用来将截取其实索引和结束索引之间的数组元素。

# 字符串与数组互转

const str = "hello world";
const strArr = str.split(" ");
console.log("split", strArr); // [ 'hello', 'world' ]
const newStr = strArr.join(" ");
console.log("join", newStr); // hello world
console.log("toString", strArr.toString()); // hello,world

案例解析

split()方法可以把字符串根据分隔符切成由字符组成的数组。join()方法可以通过分隔符把由字符组成的数组拼合成一个字符串。注意的是，如果 join()没有传入分隔符，默认会用逗号分隔字符。toString()、toLocaleString()和 join()一样，也可以将数组转换为一个字符串，但两者的区别在于，join()可以指定分隔符，但 toString()和toLocaleString()无法指定分隔符，只能以逗号分隔字符。

# 字符串拼接

const str1 = "hello";
const str2 = "world";
console.log("+", str1 + " " + str2); // hello world
console.log("concat", str1.concat(" ", str2)); // hello world
console.log("``", `${str1} ${str2}, hello china`); // hello world, hello china

案例解析

通过+拼接字符串是最原始的方法，但缺点是非常不优雅，非常繁琐，对于大量的字符串拼接，就是一场噩梦。使用字符串的 concat()也可以将多个字符拼接在一起，返回一个新的字符串，该方法比+拼接更加优雅直观一些。最为优雅的处理方法是 ES6 新增的模板字符串，通过反引号 `` 包含要拼接的字符串和变量，变量用 ${}包裹，非常直观。特别是对于大量字符串拼接、接口参数拼接和字符串换行拼接，处理起来非常强大。

说明

数组和字符串一样，也有 concat()方法，用来将数组元素或者数组拼接成一个新的数组。

# 字符串替换

const str = "hello world";
const newStr = str.replace("world", "china");
console.log("replace", newStr); // hello china
const newStr1 = str.replace(/world/, "china");
console.log("replace", newStr1); // hello china
const reg = /^(hello)\s(world)$/;
const newStr2 = str.replace(reg, "$1");
console.log("replace", newStr2); // hello

案例解析

replace()的第一个参数可以字符，也可以匹配某个字符的正则表达式，第二个参数是要替换的新字符，可以是具体的某个字符，也可以是正则表达式中的一个分组，这里的$1 代表第一个分组(hello)匹配的字符，也就是用第一个分组的字符把整个正则表达式匹配的字符串替换掉。

# 字符串遍历

const str = "yes";
for (let i = 0; i < str.length; i++) {
  console.log(str[i]);
  // y
  // e
  // s
}
for (let char of str) {
  console.log(char);
  // y
  // e
  // s
}

案例解析

for 循环和 for...of...都可以对字符串进行遍历，如果字符串中包含空格，也会被遍历出来。但是，for...of...遍历时可以识别大于 0xFFFF 的码点，for 循环不可以，这里不再追溯。

# 字符串补全

const hours = "8";
const minute = "9";
console.log(`${hours.padStart(2, 0)}: ${minute.padStart(2, 0)}`); // 08: 09
console.log(hours.padEnd(2, 8)); // 88

案例解析

padStart()用来在字符串头部补全字符，padEnd()用来在字符串尾部补全字符。第一个参数是要补全的长度，第二个参数是用来补全的字符串。最为常用的场景，就是在单数构成的时分秒前面补 0。但需要注意的是，如果需要将时间用于计时器等，需要将时间转换为数值类型。

# 字符串重复

console.log(str.repeat(1)); // yes
console.log(str.repeat(0)); // ''
console.log(str.repeat(3)); // yesyesyes
console.log(str.repeat("3")); // yesyesyes

案例解析

repeat()方法可以将字符串重复指定次数，返回一个新的字符串。主要注意的是，并不是在原有字符串的基础上增加重复的字符串，重复 0 次，会返回一个空字符串，重复 1 次，返回和原始字符串相同的字符串。如果传入的重复次数是字符串类型，会被转换为数值类型。

# 字符串反转

const str = "hello world";
function reverse(str) {
  return str
    .split("")
    .reverse()
    .join("");
}
console.log("reverse", reverse(str)); // dlrow olleh

案例解析

split()方法将字符串转换为数组，分隔符必须是空字符串。然后利用 reverse()反转数组元素，也就是字符串对象的每隔字符。最后使用 join()将数组转换为字符串，分隔符也是空字符串。

# 字符串去除空格

const str = " hello world ";
const newStr = str.trim();
console.log("trim", newStr); // 'hello world'

const str1 = " hello world ";
const newStr1 = str1.trimLeft();
console.log("trimLeft", newStr1); // 'hello world '

const str2 = " hello world ";
const newStr2 = str2.trimRight();
console.log("trimRight", newStr2); // ' hello world'

const str3 = " hello world ";
const newStr3 = str3.replace(/\s+/g, "");
console.log("replace-all", newStr3); // 'helloworld'

const str4 = " hello world ";
const newStr4 = str4.replace(/^\s*/g, "");
console.log("replace-left", newStr4); // 'hello world '

const str5 = " hello world ";
const newStr5 = str5.replace(/\s*$/g, "");
console.log("replace-right", newStr5); // ' hello world'

const str6 = " hello world ";
const newStr6 = str6.replace(/(^\s*)|(\s*$)/g, "");
console.log("replace-left-right", newStr6); // 'hello world'

案例解析

可以使用 trim()去除字符串左边和右边空格，trim()方法不会改变原始字符串，会返回一个新的字符串，所以需要赋值给一个变量。trimLeft()会去除左边的空格，trimRight()会去除右边的空格。去除全部的空格，需要使用正则表达式和 replace()方法，将正则表达式/\s+/g 匹配的全部空格替换为一个空字符串。\s 匹配的空白字符包括空格、tap 制表符、换行符等。

去除左空格、右空格和左右两边空格，也可以使用正则表达式和 replace()方法，在浏览器不支持 trim()方法是，也是种兼容处理手段。/^\s\*/g 匹配以 0 个或多个空白字符开始的字符，去除左边空格，/\s\*\$/g 匹配以 0 个或多个空白字符结尾的字符，去除右边边空格，/(^\s*)|(\s*\$)/g 匹配以 0 个或多个空白字符开始的字符或者以 0 个或多个空白字符结尾的字符，去除左右两边的空格。

# 字符串大小写转换

const str = "hello world";
const newStr = str.toUpperCase();
console.log(newStr); // HELLO WORLD
console.log(newStr.toLowerCase()); // hello world

// 首字母大写
const str1 = "hello world";
const newStr1 = str1.substring(0, 1).toUpperCase() + str1.substring(1);
// 或者：
const newStr1 = str1.replace(/^[a-z]/g, str => str.toUpperCase());
console.log(newStr1); // Hello world

// 字符串中每个单词的首字母大写
const str2 = "hello world";
const strArr = str2.split(/\s+/);
const newStrArr = [];
strArr.forEach(item => {
  item = item.substring(0, 1).toUpperCase() + item.substring(1);
  newStrArr.push(item);
});
const newStr2 = newStrArr.join(" ");
console.log(newStr2); // Hello World

const str3 = "hello world";
const strArr1 = str3.split(/\s+/);
function upperCase(strArr) {
  return strArr
    .map(item => {
      return item.slice(0, 1).toUpperCase() + item.slice(1);
    })
    .join(" ");
}
console.log("map", upperCase(strArr1)); // Hello World

const str4 = "hello world";
const newStr4 = str4.replace(/( |^)[a-z]/g, str => str.toUpperCase());
console.log("replace", newStr4); // Hello World

案例解析

全部大写或者小写，调用 toUpperCase()或者 toLowerCase()即可。首字母大写需要截取首字母并大写，然后在和后面的字符拼接成格式化后的新字符串；或者使用 replace()方法传递两个参数，/^[a-z]/g 匹配首字母，第二项是一个函数，函数参数是匹配到的首字母，返回值中调用 toUpperCase()将其大写。

字符串中每个单词的首字母大写使用了三种方式，第一种是将字符串使用 split()转为数组进行遍历，然后依然使用字符串截取和拼接来实现每个单词首字母大写。需要注意的是，这里无法直接更新原始数组，所以需要手动 push 格式化后的字符串到新数组。map() 方法和 forEach 类似，也是利用了遍历、字符串截取和拼接来实现，单这里对 map()方法的操作进行了很好的封装。因为 map()方法会返回一个新的数组，这里直接在 map()方法后面使用 join()方法将返回值转换为字符串，最后再 return 出去。当然不使用函数封装，map()需要赋值给一个变量保存处理后的新数组。最后一种方法是利用了正则表达式/( |^)[a-z]/g 匹配字母前面有空格的或者以字母开头的字符，并转换为大写，替换后的新字符串赋值给新的变量保存。
